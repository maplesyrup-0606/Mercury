## My solution

```python
class Solution:
    def equalPairs(self, grid: List[List[int]]) -> int:
        columns = {}
        for i in range(len(grid)) :
            col_i = []
            for j in range(len(grid[i])) : 
                col_i.append(grid[j][i])

            if str(col_i) in columns : 
                columns[str(col_i)] += 1
            
            else :
                columns[str(col_i)] = 1

        rows = {}
        for row in grid: 
            if str(row) in rows :
                rows[str(row)] += 1
            
            else :
                rows[str(row)] = 1
            

        count = 0    
        for col in columns :
            if col in rows :
                count += columns[col] * rows[col]
        
        return count
```
