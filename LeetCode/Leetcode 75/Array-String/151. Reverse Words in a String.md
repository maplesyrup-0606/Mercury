```python
class Solution:
    def reverseWords(self, s: str) -> str:
        
        i = 0
        res = ""
        while i < len(s) :
            
            while i < len(s) and s[i] == " " :
                i += 1
            
            temp = ""

            while i < len(s) and s[i] != " " : 
                temp += s[i]
                i += 1
            
            if temp == "" : continue
            res = temp + " " + res
        
        return res[:-1]
```
