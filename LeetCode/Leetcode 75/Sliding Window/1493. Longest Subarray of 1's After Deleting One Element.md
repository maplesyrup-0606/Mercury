```python
class Solution:
    def longestSubarray(self, nums: List[int]) -> int:
        left, ret, count = 0, 0, 0

        for i in range(len(nums)) : 
            if nums[i] == 0 :
                count += 1
            
            while count > 1 :
                if nums[left] == 0 :
                    count -= 1
                left += 1
            
            ret = max(ret, i - left + 1 - count)
    
        return ret - 1 if ret == len(nums) else ret
```
