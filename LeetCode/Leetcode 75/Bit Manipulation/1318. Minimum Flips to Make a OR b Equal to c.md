```python
class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        top = math.floor(math.log(max(a,b,c), 2))
        res = 0

        while top >= 0 :
            a_ = a >> top & 1
            b_ = b >> top & 1
            c_ = c >> top & 1
            if a_ | b_ != c_ :
                if c_ == 1: # a or b == 0  and c == 1
                    res += 1
                else : # a or b == 1 and c == 0
                    if a_ == 1 and b_ == 1 :
                        res += 2
                    
                    else : 
                        res += 1
            top -= 1
        return res
```
