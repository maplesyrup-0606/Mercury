## My solution

```python
class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        self.res = []
        
        def dfs(curr, n, k, nums) :
            if n == 0 :
                if k == 0 :
                    self.res.append(nums)
            
            elif n < 0 :
                return
            
            else :
                for i in range(curr - 1, 0, -1) :
                    dfs(i, n - i, k - 1, nums.copy() + [i])

        for i in range(1, 10) :
            dfs(i, n - i, k - 1, [i])
        
        return self.res
```
